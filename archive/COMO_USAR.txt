💡 EJEMPLOS DE USO
═══════════════════════════════════════════════════════════════════════

DESDE PYTHON (EN TU CÓDIGO):

    from database_manager import DatabaseManager
    from device_manager import DeviceManager
    
    # Inicializar
    db = DatabaseManager("trading_system.db")
    device_mgr = DeviceManager(db)
    
    # Registrar cliente
    db.register_customer("user1", "user@email.com", "password", "premium")
    
    # Verificar credenciales
    success, cid, msg = db.verify_customer_password("user1", "password")
    
    # Registrar dispositivo
    success, device_info = device_mgr.register_new_device(
        customer_id=cid,
        device_name="Mi Dispositivo",
        ip_address="192.168.1.1"
    )
    
    # Crear sesión
    success, token = device_mgr.create_device_session(
        customer_id=cid,
        device_fingerprint=device_info['fingerprint'],
        ip_address="192.168.1.1"
    )
    
    # Verificar sesión
    valid, customer_id = db.verify_session_token(token)
    
    # Registrar pago
    db.record_payment(cid, 99.99, "stripe", "pi_123")

🔍 TROUBLESHOOTING
═══════════════════════════════════════════════════════════════════════

   → Limpiar: python -c "import os; [os.remove(f) for f in __import__('glob').glob('*.db*')]"
   → Ejecutar: python run_system.py

❌ Puerto 5000 en uso
   → Cambiar en start.py: dashboard.run(port=8000)

❌ Módulos no encontrados
   → Instalar: pip install -r requirements.txt
